i = 2
j = 1

while i <= 9:

  if i % 2 == 1:
    if i == 3:
      print("**홀수**")
    while j <= 9:
      print("{0} X {1} = {2}".format(i, j, i*j))
      j += 2

  i += 1
  j = 1

i = 2
j = 1

while i <= 9:

  if i % 2 == 0:
    if i == 2:
      print("**짝수**")
    while j <= 9:
      if j % 2 == 0:
        print("{0} X {1} = {2}".format(i, j, i*j))
      j += 1

  i += 1
  j = 1
